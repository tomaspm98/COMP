import java.lang.StringBuilder;
import io;

class InvokeVirtual {

    public static void main(String[] args) {
        StringBuilder sb;
        StringBuilder sb2;
        String res;

        sb = new StringBuilder();

        sb.appendCodePoint(65);
        sb = sb.appendCodePoint(66);
        sb2 = sb.appendCodePoint(67);

        res = sb.toString();
        io.println(res);

        res = sb2.toString();
        io.println(res);
    }
}
